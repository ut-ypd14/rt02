/**
 * LINE_Bot.gs - 巡檢人員點檢確認表{記錄}
 * Version: v0.1.2
 *
 * 主 GAS 呼叫方式：
 *   無 action 的 POST → return lineBot_doPost_(e)
 *
 * 任務：
 *   1. 接收 LINE webhook JSON
 *   2. 寫入 lineBot_log，逐段確認流程到達位置
 *   3. 偵測「被 @」且文字含 LINE_BIND_KEYWORD
 *   4. 記錄 group / room / user 目標 ID 與觸發者 ID 到 lineBot 表
 *   5. 立即 push 一則測試訊息，驗證 target_id 可正常發送
 *
 * 注意：
 *   - target ID 不存 PropertiesService，寫入工作簿 lineBot 表。
 *   - Channel access token 仍需放 Script Properties：LINE_TEST_TOKEN。
 *   - 不會把 token 寫入 log。
 */

const LINE_BOT_MODULE_VERSION = 'v0.1.2';
const LINE_BOT_SHEET_NAME = 'lineBot';
const LINE_BOT_LOG_SHEET_NAME = 'lineBot_log';
const LINE_BIND_KEYWORD = 'robot pass';
const LINE_REQUIRE_MENTION = true;

// 初測時保留 true：若 webhook mention 物件沒有 isSelf，可用文字中含 @ 作為輔助判斷。
// 若後續要嚴格只接受「真的 @ Bot」，可改為 false，並視需要設定 Script Properties：LINE_BOT_USER_ID。
const LINE_ALLOW_AT_TEXT_FALLBACK = true;

const LINE_TOKEN_PROPERTY_NAME = 'LINE_TEST_TOKEN';
const LINE_BOT_USER_ID_PROPERTY_NAME = 'LINE_BOT_USER_ID';
const LINE_LOG_RAW_MAX = 800;
const LINE_LOG_DETAIL_MAX = 1500;

const LINE_BOT_HEADERS = [
  'target_type',
  'target_id',
  'display_name',
  'trigger_user_id',
  'trigger_user_name',
  'message_text',
  'active',
  'last_seen_at',
  'updated_at',
  'push_test_status',
  'push_test_at',
  'note'
];

const LINE_BOT_LOG_HEADERS = [
  'time',
  'stage',
  'detail',
  'note'
];

/**
 * 主 GAS 無 action POST 會進入這裡。
 */
function lineBot_doPost_(e) {
  lineBot_log_('START', 'lineBot_doPost_ entered', 'module=' + LINE_BOT_MODULE_VERSION);

  try {
    const raw = e && e.postData && e.postData.contents
      ? String(e.postData.contents || '')
      : '';

    lineBot_log_('RAW', raw ? lineBot_clip_(raw, LINE_LOG_RAW_MAX) : 'empty_raw', '');

    if (!raw) {
      lineBot_log_('STOP', 'empty_raw', '');
      return lineBot_text_('line_bot_empty');
    }

    let data;
    try {
      data = JSON.parse(raw);
      lineBot_log_('JSON', 'parse_ok', '');
    } catch (err) {
      lineBot_log_('JSON_ERROR', String(err && err.message ? err.message : err), '');
      return lineBot_text_('line_bot_invalid_json');
    }

    const events = Array.isArray(data.events) ? data.events : [];
    lineBot_log_('EVENTS', 'count=' + events.length, '');

    if (!events.length) {
      lineBot_log_('END', 'line_bot_no_events', '');
      return lineBot_text_('line_bot_no_events');
    }

    const result = {
      total: events.length,
      processed: 0,
      skipped: 0,
      errors: 0
    };

    events.forEach(function(event, index) {
      try {
        const one = lineBot_handleEvent_(event, index);

        if (one && one.processed) {
          result.processed += 1;
        } else {
          result.skipped += 1;
        }
      } catch (err) {
        result.errors += 1;
        lineBot_log_(
          'EVENT_ERROR_' + index,
          String(err && err.message ? err.message : err),
          ''
        );
      }
    });

    lineBot_log_('RESULT', JSON.stringify(result), '');
    lineBot_log_('END', 'line_bot_ok', '');
    return lineBot_text_('ok');

  } catch (err) {
    lineBot_log_('SYSTEM_ERROR', String(err && err.message ? err.message : err), '');
    return lineBot_text_('line_bot_error');
  }
}

function lineBot_handleEvent_(event, index) {
  const idx = Number(index || 0);
  const eventType = String(event && event.type ? event.type : '');
  const source = event && event.source ? event.source : {};
  const message = event && event.message ? event.message : {};
  const text = String(message.text || '');

  lineBot_log_(
    'EVENT_' + idx,
    [
      'type=' + eventType,
      'sourceType=' + String(source.type || ''),
      'groupId=' + String(source.groupId || ''),
      'roomId=' + String(source.roomId || ''),
      'userId=' + String(source.userId || ''),
      'messageType=' + String(message.type || ''),
      'text=' + text
    ].join(' | '),
    ''
  );

  if (!event || eventType !== 'message') {
    return lineBot_skip_(idx, 'not_message_event');
  }

  if (message.type !== 'text') {
    return lineBot_skip_(idx, 'not_text_message');
  }

  const keywordOk = lineBot_textHasKeyword_(text);
  lineBot_log_('CHECK_KEYWORD_' + idx, 'keyword=' + LINE_BIND_KEYWORD + ' | ok=' + keywordOk, '');

  if (!keywordOk) {
    return lineBot_skip_(idx, 'keyword_not_found');
  }

  const mentionOk = !LINE_REQUIRE_MENTION || lineBot_isMentioned_(event, text, idx);
  lineBot_log_(
    'CHECK_MENTION_' + idx,
    'require=' + LINE_REQUIRE_MENTION + ' | ok=' + mentionOk,
    ''
  );

  if (!mentionOk) {
    return lineBot_skip_(idx, 'mention_not_found');
  }

  const target = lineBot_getTarget_(source);
  lineBot_log_(
    'TARGET_' + idx,
    'targetType=' + target.targetType + ' | targetId=' + target.targetId,
    ''
  );

  if (!target.targetId) {
    return lineBot_skip_(idx, 'missing_target_id');
  }

  const nowText = lineBot_nowText_();
  const triggerUserId = String(source.userId || '');

  const displayName = lineBot_getTargetDisplayName_(target.targetType, target.targetId, idx);
  const triggerUserName = lineBot_getTriggerUserName_(target.targetType, target.targetId, triggerUserId, idx);

  const rowObj = {
    target_type: target.targetType,
    target_id: target.targetId,
    display_name: displayName,
    trigger_user_id: triggerUserId,
    trigger_user_name: triggerUserName,
    message_text: text,
    active: 'TRUE',
    last_seen_at: nowText,
    updated_at: nowText,
    push_test_status: '',
    push_test_at: '',
    note: 'bind_request; module=' + LINE_BOT_MODULE_VERSION
  };

  lineBot_log_('UPSERT_BEFORE_PUSH_' + idx, target.targetType + ' | ' + target.targetId, '');
  const upsert = lineBot_upsertTarget_(rowObj);
  lineBot_log_('UPSERT_OK_' + idx, 'mode=' + upsert.mode + ' | row=' + upsert.row, '');

  const testMsg = lineBot_buildBindTestMessage_(rowObj, upsert.mode);
  lineBot_log_('PUSH_START_' + idx, 'to=' + target.targetType + ' | ' + target.targetId, '');

  const push = lineBot_pushText_(target.targetId, testMsg);
  lineBot_log_(
    'PUSH_RESULT_' + idx,
    'ok=' + push.ok + ' | code=' + push.code + ' | body=' + lineBot_clip_(push.body, 300),
    ''
  );

  rowObj.push_test_status = push.ok ? 'OK' : ('ERROR_' + push.code);
  rowObj.push_test_at = lineBot_nowText_();
  rowObj.note = push.ok
    ? ('bind_ok; mode=' + upsert.mode + '; module=' + LINE_BOT_MODULE_VERSION)
    : ('bind_saved_but_push_failed; mode=' + upsert.mode + '; code=' + push.code + '; body=' + lineBot_clip_(push.body, 120));

  const upsertAfterPush = lineBot_upsertTarget_(rowObj);
  lineBot_log_('UPSERT_AFTER_PUSH_' + idx, 'mode=' + upsertAfterPush.mode + ' | row=' + upsertAfterPush.row, '');

  return {
    processed: true,
    target_type: target.targetType,
    target_id: target.targetId,
    push_ok: push.ok
  };
}

function lineBot_skip_(index, reason) {
  lineBot_log_('SKIP_' + index, String(reason || ''), '');
  return {
    processed: false,
    reason: String(reason || '')
  };
}

function lineBot_textHasKeyword_(text) {
  const t = String(text || '').toLowerCase();
  const k = String(LINE_BIND_KEYWORD || '').toLowerCase();

  if (!k) return false;
  return t.indexOf(k) >= 0;
}

function lineBot_isMentioned_(event, text, index) {
  const message = event && event.message ? event.message : {};
  const mention = message.mention || {};
  const mentionees = Array.isArray(mention.mentionees) ? mention.mentionees : [];
  const botUserId = lineBot_getBotUserId_();
  let detail = 'mentionees=' + mentionees.length + ' | botUserId_set=' + !!botUserId;

  for (let i = 0; i < mentionees.length; i++) {
    const m = mentionees[i] || {};
    detail += ' | m' + i + '.isSelf=' + String(m.isSelf || false);
    detail += ' | m' + i + '.userId=' + String(m.userId || '');

    if (m.isSelf === true) {
      lineBot_log_('MENTION_DETAIL_' + index, detail + ' | result=isSelf', '');
      return true;
    }

    if (botUserId && String(m.userId || '') === botUserId) {
      lineBot_log_('MENTION_DETAIL_' + index, detail + ' | result=botUserId_match', '');
      return true;
    }
  }

  if (LINE_ALLOW_AT_TEXT_FALLBACK && String(text || '').indexOf('@') >= 0) {
    lineBot_log_('MENTION_DETAIL_' + index, detail + ' | result=at_text_fallback', '');
    return true;
  }

  lineBot_log_('MENTION_DETAIL_' + index, detail + ' | result=false', '');
  return false;
}

function lineBot_getTarget_(source) {
  const type = String(source && source.type ? source.type : '').toLowerCase();

  if (type === 'group') {
    return {
      targetType: 'group',
      targetId: String(source.groupId || '')
    };
  }

  if (type === 'room') {
    return {
      targetType: 'room',
      targetId: String(source.roomId || '')
    };
  }

  if (type === 'user') {
    return {
      targetType: 'user',
      targetId: String(source.userId || '')
    };
  }

  return {
    targetType: type || '',
    targetId: ''
  };
}

function lineBot_getTargetDisplayName_(targetType, targetId, index) {
  if (!targetId) return '';

  if (targetType === 'group') {
    const summary = lineBot_getGroupSummary_(targetId);
    lineBot_log_(
      'GROUP_SUMMARY_' + index,
      'ok=' + summary.ok + ' | code=' + summary.code + ' | groupName=' + summary.groupName,
      summary.ok ? '' : lineBot_clip_(summary.body, 300)
    );

    if (summary.ok && summary.groupName) return summary.groupName;
    return 'LINE群組';
  }

  if (targetType === 'room') {
    return 'LINE多人聊天室';
  }

  if (targetType === 'user') {
    const profile = lineBot_getUserProfile_(targetId);
    lineBot_log_(
      'USER_PROFILE_' + index,
      'ok=' + profile.ok + ' | code=' + profile.code + ' | displayName=' + profile.displayName,
      profile.ok ? '' : lineBot_clip_(profile.body, 300)
    );

    if (profile.ok && profile.displayName) return profile.displayName;
    return 'LINE使用者';
  }

  return '';
}

function lineBot_getTriggerUserName_(targetType, targetId, userId, index) {
  if (!userId) return '';

  if (targetType === 'group' && targetId) {
    const p1 = lineBot_getGroupMemberProfile_(targetId, userId);
    lineBot_log_(
      'TRIGGER_GROUP_PROFILE_' + index,
      'ok=' + p1.ok + ' | code=' + p1.code + ' | displayName=' + p1.displayName,
      p1.ok ? '' : lineBot_clip_(p1.body, 300)
    );

    if (p1.ok && p1.displayName) return p1.displayName;
  }

  if (targetType === 'room' && targetId) {
    const p2 = lineBot_getRoomMemberProfile_(targetId, userId);
    lineBot_log_(
      'TRIGGER_ROOM_PROFILE_' + index,
      'ok=' + p2.ok + ' | code=' + p2.code + ' | displayName=' + p2.displayName,
      p2.ok ? '' : lineBot_clip_(p2.body, 300)
    );

    if (p2.ok && p2.displayName) return p2.displayName;
  }

  const p3 = lineBot_getUserProfile_(userId);
  lineBot_log_(
    'TRIGGER_USER_PROFILE_' + index,
    'ok=' + p3.ok + ' | code=' + p3.code + ' | displayName=' + p3.displayName,
    p3.ok ? '' : lineBot_clip_(p3.body, 300)
  );

  if (p3.ok && p3.displayName) return p3.displayName;

  return '';
}

function lineBot_getGroupSummary_(groupId) {
  const res = lineBot_apiFetch_(
    'https://api.line.me/v2/bot/group/' + encodeURIComponent(groupId) + '/summary',
    'get',
    null
  );

  if (!res.ok) {
    return {
      ok: false,
      code: res.code,
      groupId: groupId,
      groupName: '',
      pictureUrl: '',
      body: res.body
    };
  }

  const data = lineBot_safeJson_(res.body);

  return {
    ok: true,
    code: res.code,
    groupId: String(data.groupId || groupId),
    groupName: String(data.groupName || ''),
    pictureUrl: String(data.pictureUrl || '')
  };
}

function lineBot_getGroupMemberProfile_(groupId, userId) {
  return lineBot_profileFetch_(
    'https://api.line.me/v2/bot/group/' + encodeURIComponent(groupId) +
    '/member/' + encodeURIComponent(userId)
  );
}

function lineBot_getRoomMemberProfile_(roomId, userId) {
  return lineBot_profileFetch_(
    'https://api.line.me/v2/bot/room/' + encodeURIComponent(roomId) +
    '/member/' + encodeURIComponent(userId)
  );
}

function lineBot_getUserProfile_(userId) {
  return lineBot_profileFetch_(
    'https://api.line.me/v2/bot/profile/' + encodeURIComponent(userId)
  );
}

function lineBot_profileFetch_(url) {
  const res = lineBot_apiFetch_(url, 'get', null);

  if (!res.ok) {
    return {
      ok: false,
      code: res.code,
      displayName: '',
      userId: '',
      pictureUrl: '',
      statusMessage: '',
      body: res.body
    };
  }

  const data = lineBot_safeJson_(res.body);

  return {
    ok: true,
    code: res.code,
    displayName: String(data.displayName || ''),
    userId: String(data.userId || ''),
    pictureUrl: String(data.pictureUrl || ''),
    statusMessage: String(data.statusMessage || '')
  };
}

function lineBot_pushText_(to, text) {
  return lineBot_apiFetch_(
    'https://api.line.me/v2/bot/message/push',
    'post',
    {
      to: to,
      messages: [
        {
          type: 'text',
          text: String(text || '')
        }
      ]
    }
  );
}

function lineBot_apiFetch_(url, method, payload) {
  try {
    const token = lineBot_getToken_();

    const options = {
      method: method,
      headers: {
        Authorization: 'Bearer ' + token
      },
      muteHttpExceptions: true
    };

    if (payload !== null && payload !== undefined) {
      options.contentType = 'application/json';
      options.payload = JSON.stringify(payload);
    }

    const res = UrlFetchApp.fetch(url, options);
    const code = res.getResponseCode();
    const body = res.getContentText();

    return {
      ok: code >= 200 && code < 300,
      code: code,
      body: body
    };

  } catch (err) {
    return {
      ok: false,
      code: 0,
      body: 'exception:' + String(err && err.message ? err.message : err)
    };
  }
}

function lineBot_buildBindTestMessage_(rowObj, mode) {
  const targetName = String(rowObj.display_name || '').trim() || '未取得目標名稱';
  const triggerName = String(rowObj.trigger_user_name || '').trim() || '沒有成員';

  return [
    'LINE 綁定測試成功',
    '版本：' + LINE_BOT_MODULE_VERSION,
    '模式：' + mode,
    '類型：' + rowObj.target_type,
    '目標：' + targetName,
    '觸發者：' + triggerName,
    '時間：' + lineBot_nowText_()
  ].join('\n');
}

function lineBot_upsertTarget_(rowObj) {
  const sh = lineBot_getSheet_();
  const targetType = String(rowObj.target_type || '');
  const targetId = String(rowObj.target_id || '');

  if (!targetType || !targetId) {
    throw new Error('missing_line_target_key');
  }

  const values = [
    targetType,
    targetId,
    String(rowObj.display_name || ''),
    String(rowObj.trigger_user_id || ''),
    String(rowObj.trigger_user_name || ''),
    String(rowObj.message_text || ''),
    String(rowObj.active || 'TRUE'),
    String(rowObj.last_seen_at || ''),
    String(rowObj.updated_at || ''),
    String(rowObj.push_test_status || ''),
    String(rowObj.push_test_at || ''),
    String(rowObj.note || '')
  ];

  const rowNo = lineBot_findTargetRow_(sh, targetType, targetId);

  if (rowNo > 0) {
    sh.getRange(rowNo, 1, 1, values.length).setValues([values]);
    return {
      mode: 'update',
      row: rowNo
    };
  }

  sh.appendRow(values);

  return {
    mode: 'insert',
    row: sh.getLastRow()
  };
}

function lineBot_getSheet_() {
  const ss = lineBot_getSS_();
  let sh = ss.getSheetByName(LINE_BOT_SHEET_NAME);

  if (!sh) {
    sh = ss.insertSheet(LINE_BOT_SHEET_NAME);
  }

  const headerRange = sh.getRange(1, 1, 1, LINE_BOT_HEADERS.length);
  const current = headerRange.getValues()[0];
  let needHeader = false;

  for (let i = 0; i < LINE_BOT_HEADERS.length; i++) {
    if (String(current[i] || '') !== LINE_BOT_HEADERS[i]) {
      needHeader = true;
      break;
    }
  }

  if (needHeader) {
    headerRange.setValues([LINE_BOT_HEADERS]);
    sh.setFrozenRows(1);
  }

  return sh;
}

function lineBot_findTargetRow_(sh, targetType, targetId) {
  const lastRow = sh.getLastRow();

  if (lastRow < 2) return 0;

  const values = sh.getRange(2, 1, lastRow - 1, 2).getValues();

  for (let i = values.length - 1; i >= 0; i--) {
    if (String(values[i][0] || '') === targetType && String(values[i][1] || '') === targetId) {
      return i + 2;
    }
  }

  return 0;
}

function lineBot_log_(stage, detail, note) {
  try {
    const ss = lineBot_getSS_();
    let sh = ss.getSheetByName(LINE_BOT_LOG_SHEET_NAME);

    if (!sh) {
      sh = ss.insertSheet(LINE_BOT_LOG_SHEET_NAME);
      sh.getRange(1, 1, 1, LINE_BOT_LOG_HEADERS.length).setValues([LINE_BOT_LOG_HEADERS]);
      sh.setFrozenRows(1);
    }

    const nowText = lineBot_nowText_();

    sh.appendRow([
      nowText,
      String(stage || ''),
      lineBot_clip_(String(detail || ''), LINE_LOG_DETAIL_MAX),
      lineBot_clip_(String(note || ''), LINE_LOG_DETAIL_MAX)
    ]);

  } catch (err) {
    Logger.log('lineBot_log_ failed: ' + String(err && err.message ? err.message : err));
  }
}

function lineBot_getSS_() {
  if (typeof getSS_ === 'function') {
    return getSS_();
  }

  if (typeof SPREADSHEET_ID !== 'undefined' && SPREADSHEET_ID) {
    return SpreadsheetApp.openById(SPREADSHEET_ID);
  }

  throw new Error('missing_getSS_or_SPREADSHEET_ID');
}

function lineBot_getToken_() {
  const token = PropertiesService
    .getScriptProperties()
    .getProperty(LINE_TOKEN_PROPERTY_NAME);

  if (!token) {
    throw new Error('missing_' + LINE_TOKEN_PROPERTY_NAME);
  }

  return token;
}

function lineBot_getBotUserId_() {
  return String(PropertiesService
    .getScriptProperties()
    .getProperty(LINE_BOT_USER_ID_PROPERTY_NAME) || '');
}

function lineBot_nowText_() {
  return Utilities.formatDate(new Date(), lineBot_tz_(), 'yyyy/MM/dd HH:mm:ss');
}

function lineBot_tz_() {
  if (typeof APP_TZ !== 'undefined' && APP_TZ) {
    return APP_TZ;
  }

  return 'Asia/Taipei';
}

function lineBot_safeJson_(text) {
  try {
    return JSON.parse(String(text || '{}'));
  } catch (err) {
    return {};
  }
}

function lineBot_clip_(text, maxLen) {
  const s = String(text || '');
  const n = Number(maxLen || 120);

  if (s.length <= n) return s;
  return s.slice(0, n) + '...';
}

function lineBot_text_(text) {
  return ContentService
    .createTextOutput(String(text || 'ok'))
    .setMimeType(ContentService.MimeType.TEXT);
}
